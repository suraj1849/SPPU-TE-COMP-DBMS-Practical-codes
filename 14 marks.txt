CREATE OR REPLACE PROCEDURE Proc_Grade (
	pMarks IN NUMBER,
	pClass OUT varchar
) AS
BEGIN
	IF pMarks >= 990 AND pMarks <= 1500 THEN
		pClass := 'Destinction';
	ELSIF pMarks >= 900 AND pMarks <= 989 THEN
		pClass := 'First Class';
	ELSIF pMarks >= 825 AND pMarks <= 899 THEN
		pClass := 'Second Class';
	ELSE
		pClass := 'Fail';
	END IF;
END;
